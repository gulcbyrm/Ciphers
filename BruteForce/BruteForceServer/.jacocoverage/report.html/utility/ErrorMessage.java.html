package utility;



public enum ErrorMessage {

    NO_ERROR(0, "...  HATAYLA  KARŞILAŞILMADI ..."),
    NULL_ENTITY(1, "Nesne NULL. Talep edilen işlemi iptal etmek zorunda kaldık..."),
    MAX_LENGTH(3, "Belirtilen kolon MAX değeri aşıyor. Talep edilen işlemi iptal etmek zorunda kaldık..."),
    RUNTIMEEXCEPTION(5, " işlem yapılırken RuntimeException hatası oluştu "),
    CANNOT_CONVERT(16, "Tip Dönüştürme Hatası"),
    EMPTY(18, "Beklenen veri/veriler girilmedi. Boş kayıt veya verilerin bir kısmını eksik girdiniz"),
        IO_FORM_LOAD(10, "Form açılmaya çalışılırken giriş çıkış hatası oluştu"),

    ANOTHER_ERROR(21, "İşlem yapılırken tanımlanamayan bir hata meydana geldi");


    private final int errorNumber;
    private final String errorMessages;

    ErrorMessage(int errorNumber, String errorMessages) {
        this.errorNumber = errorNumber;
        this.errorMessages = errorMessages;

    }

    public boolean isError() {
        return this.getErrorNumber() > 0;
    }

    public int getErrorNumber() {
        return errorNumber;
    }

    public String getErrorMessages() {
        return errorMessages;
    }


    public void printErrorMessages() {
        new MyAlert().showErrorAlert( errorMessages, "HATA");
    }


    public void printErrorMessages(String exMessage) {
        new MyAlert().showErrorAlert(errorMessages+"\n" +exMessage,"HATA OLUŞTU");
    }


    public void printErrorMessages(String exMessage,Exception e) {
        new MyAlert().showErrorAlert(errorMessages+"\n" +exMessage+"\n"+e.getMessage(),"HATA OLUŞTU");
        //e.printStackTrace();
    }
}
